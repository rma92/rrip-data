# NYC Zip Code ranges
Manhattan: 10001–10292
Brooklyn: 11201–11239
Queens: 11004–11005, 11101–11120, 11351–11697
Bronx: 10451–10475
Staten Island: 10301–10314

# Get only NYC Voters
```sqlite
PRAGMA synchronous=OFF;
PRAGMA count_changes=OFF;
PRAGMA journal_mode=OFF;
PRAGMA temp_store=OFF;
ATTACH DATABASE 'NY_voters.DB' AS V;
CREATE TABLE V(
  LASTNAME TEXT,
  FIRSTNAME TEXT,
  MIDDLENAME TEXT,
  NAMESUFFIX TEXT,
  RADDNUMBER TEXT,
  RHALFCODE TEXT,
  RPREDIRECTION TEXT,
  RSTREETNAME TEXT,
  RPOSTDIRECTION TEXT,
  RAPARTMENTTYPE TEXT,
  RAPARTMENT TEXT,
  RADDRNONSTD TEXT,
  RCITY TEXT,
  RZIP5 INT,
  RZIP4 INT,
  DOB INT,
  GENDER TEXT,
  PARTY TEXT,
  OTHERPARTY TEXT,
  COUNTYCODE TEXT,
  ED INT,
  LD INT,
  TOWNCITY TEXT,
  WARD TEXT,
  CD INT,
  SD INT,
  AD INT,
  LASTVOTERDATE INT,
  STATUS TEXT,
  REASONCODE TEXT,
  INACT_DATE INT,
  PURGE_DATE INT,
  SBOEID TEXT
);
INSERT INTO V 
SELECT *
FROM V.V
WHERE
    -- Manhattan
    (rzip5 BETWEEN '10001' AND '10292')
    
    -- Brooklyn
    OR (rzip5 BETWEEN '11201' AND '11239')
    
    -- Queens
    OR (rzip5 BETWEEN '11004' AND '11005')
    OR (rzip5 BETWEEN '11101' AND '11120')
    OR (rzip5 BETWEEN '11351' AND '11697')
    
    -- Bronx
    OR (rzip5 BETWEEN '10451' AND '10475')
    
    -- Staten Island
    OR (rzip5 BETWEEN '10301' AND '10314');
```
